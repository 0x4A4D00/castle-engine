!!ARBfp1.0

# Taken from
# http://en.wikipedia.org/wiki/ARB_%28GPU_assembly_language%29
# What does this do ?
#   color.rgb := 1 - abs(1 - fragment.texcoord[0].y * 2)
# So texture coord 0 gives 0, 1 gives 0, 0.5 gives 1, and interpolates smoothly
# between.

TEMP color;
MUL color, fragment.texcoord[0].y, 2.0;
ADD color, 1.0, -color;
ABS color, color;
ADD result.color, 1.0, -color;
MOV result.color.a, 1.0;
END
